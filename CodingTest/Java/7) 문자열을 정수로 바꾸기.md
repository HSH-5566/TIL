# 문제 설명

# 풀이 1

<pre>
<code>
class Solution {
    public int solution(String s) {
        int result;
        return result = (s.charAt(0) == '-')? Integer.parseInt(s.substring(1)) * -1: Integer.parseInt(s);
    }
}

class Solution {
    public int solution(String s) {
        boolean Sign = true;
            int result = 0;

      for (int i = 0; i < s.length(); i++) {
                char ch = s.charAt(i);
                if (ch == '-')
                    Sign = false;
                else if(ch !='+')
                    result = result * 10 + (ch - '0');
            }
            return Sign?result:-1 * result;
    }
}
